/** * Perbaikan Struktur Kode Kalkulator Invila™ * Fokus: PDAM vs Jasa Invila (ABT Dihapus) */ function updateSavings() { // 1. Ambil nilai input dari elemen HTML const volTotal = parseFloat(document.getElementById('vol_total').value) || 0; const pdamPrice = parseFloat(document.getElementById('pdam_price').value) || 0; const invilaPrice = parseFloat(document.getElementById('invila_price').value) || 0; // 2. Logika Perhitungan // Target yield reklamasi ditetapkan 80% (0.8) const yieldRate = 0.8; const volReclaimed = volTotal * yieldRate; // Menghitung selisih harga per meter kubik const savingPerCubic = pdamPrice - invilaPrice; const dailySaving = volReclaimed * savingPerCubic; // Menghitung bulanan (basis 31 hari) dan tahunan const monthlySaving = dailySaving * 31; const annualSaving = monthlySaving * 12; // 3. Formatter Rupiah (IDR) const formatter = new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', minimumFractionDigits: 0, maximumFractionDigits: 0 }); // 4. Update Tampilan ke UI const monthlyEl = document.getElementById('monthly_saving'); const annualEl = document.getElementById('annual_saving'); // Gunakan pengecekan elemen agar script tidak error jika ID tidak ditemukan if (monthlyEl) { monthlyEl.innerText = formatter.format(monthlySaving).replace("Rp", "Rp "); } if (annualEl) { annualEl.innerText = formatter.format(annualSaving).replace("Rp", "Rp ") + " / Tahun"; } } function kirimWhatsApp() { // Mengambil teks hasil akhir dari layar const monthly = document.getElementById('monthly_saving')?.innerText || "0"; const annual = document.getElementById('annual_saving')?.innerText || "0"; const message = `Halo Tim Invila, saya telah melakukan simulasi kalkulator.\n\n` + `Estimasi penghematan biaya air PDAM kami adalah:\n` + `- Per Bulan: ${monthly}\n` + `- Per Tahun: ${annual}\n\n` + `Mohon info lebih lanjut mengenai survey site.`; // Pastikan nomor tujuan sesuai dengan kebutuhan Anda const waUrl = `https://wa.me/628123456789?text=${encodeURIComponent(message)}`; window.open(waUrl, '_blank'); } // Menghubungkan semua elemen agar otomatis menghitung saat diketik document.querySelectorAll('input').forEach(input => { input.addEventListener('input', updateSavings); }); // Menjalankan kalkulasi pertama kali saat halaman selesai dimuat (Initial Load) document.addEventListener('DOMContentLoaded', updateSavings);
